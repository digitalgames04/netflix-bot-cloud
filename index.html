const express = require('express'); const dotenv = require('dotenv'); dotenv.config(); console.log("Valor de GOOGLE_CREDENTIALS (antes de importar validateClient):", process.env.GOOGLE_CREDENTIALS); console.log("Valor de TEST_VAR:", process.env.TEST_VAR); const validateClient = require('./validateClient'); const app = express(); app.use(express.json()); app.post('/webhook', async (req, res) => { const { from, body } = req.body; const message = body?.entry?.[0]?.changes?.[0]?.value?.messages?.[0]?.text?.body; if (!message) return res.sendStatus(400); const clientStatus = await validateClient(from); if (!clientStatus.active) { return res.send({ message: `🚫 Tu número no está activo. Contacta a soporte: ${process.env.SUPPORT_LINK}` }); } return res.send({ message: `✅ Hola ${clientStatus.name}, ¿cómo podemos ayudarte hoy?` }); }); app.get('/', (req, res) => { res.send('Bot Netflix funcionando'); }); app.listen(process.env.PORT || 3000, () => { console.log('Bot corriendo en puerto ' + (process.env.PORT || 3000)); });
